389inline std::vector<SpacedSeed>
390parse_seeds(const std::vector<std::string>& seed_strings)
391{
392 std::vector<SpacedSeed> seed_set;
393 for (const auto& seed_string : seed_strings) {
394 SpacedSeed seed;
395 size_t pos = 0;
396 for (const auto& c : seed_string) {
397 if (c != '1') {
398 seed.push_back(pos);
399 }
400 ++pos;
401 }
402 seed_set.push_back(seed);
403 }
404 return seed_set;
405}

407inline std::vector<SpacedSeed>
408parse_blocks(const std::vector<std::string>& seed_strings)
409{
410 std::vector<SpacedSeed> blocks;
411 for (const auto& seed_string : seed_strings) {
412 const std::string padded_string = '0' + seed_string + '0';
413 SpacedSeed seed;
414 unsigned i_start = 0;
415 bool is_block = false;
416 for (unsigned pos = 0; pos < padded_string.length(); pos++) {
417 if (padded_string[pos] == '0' && is_block) {
418 seed.push_back(i_start - 1);
419 seed.push_back(pos - 1);
420 is_block = false;
421 } else if (padded_string[pos] == '1' && !is_block) {
422 i_start = pos;
423 is_block = true;
424 }
425 }
426 blocks.push_back(seed);
427 }
428 return blocks;
429}

431inline std::vector<SpacedSeed>
432parse_blocks(const std::vector<SpacedSeed>& seeds, const unsigned k)
433{ // TODO: simple but naive implementation; to be optimized
434 std::vector<std::string> seed_strings;
435 for (const SpacedSeed& seed : seeds) {
436 std::string seed_string(k, '1');
437 for (const auto& i : seed) {
438 seed_string[i] = '0';
439 }
440 seed_strings.push_back(seed_string);
441 }
442 return parse_blocks(seed_strings);

461 inline bool CLASS::init() \
462 { \
463 if (MEMBER_PREFIX k > MEMBER_PREFIX seq_len) { \
464 MEMBER_PREFIX pos = std::numeric_limits<std::size_t>::max(); \
465 return false; \
466 } \
467 unsigned posN = 0; \
468 while ( \
469 (MEMBER_PREFIX pos < MEMBER_PREFIX seq_len - MEMBER_PREFIX k + 1) && \
470 !(NTHASH_CALL)) { \
471 MEMBER_PREFIX pos += posN + 1; \
472 } \
473 if (MEMBER_PREFIX pos > MEMBER_PREFIX seq_len - MEMBER_PREFIX k) { \
474 MEMBER_PREFIX pos = std::numeric_limits<std::size_t>::max(); \
475 return false; \

478 return true; \

482#define BTLLIB_NTHASH_ROLL(CLASS, NTHASH_CALL, MEMBER_PREFIX) \
483 inline bool CLASS::roll() \
484 { \
485 if (!MEMBER_PREFIX initialized) { \
486 return init(); \
487 } \
488 if (MEMBER_PREFIX pos >= MEMBER_PREFIX seq_len - MEMBER_PREFIX k) { \
489 return false; \
490 } \
491 if (SEED_TAB[(unsigned char)(MEMBER_PREFIX seq[MEMBER_PREFIX pos + \
492 MEMBER_PREFIX k])] == \
493 SEED_N) { \
494 MEMBER_PREFIX pos += MEMBER_PREFIX k; \
495 return init(); \
496 } \
497 (NTHASH_CALL); \
498 ++MEMBER_PREFIX pos; \
499 return true; \
500 }

502// NOLINTNEXTLINE
503#define BTLLIB_NTHASH_ROLL_BACK(CLASS, NTHASH_CALL, MEMBER_PREFIX) \
504 inline bool CLASS::roll_back() \
505 { \
506 if (!MEMBER_PREFIX initialized) { \
507 return init(); \
508 } \
509 if (MEMBER_PREFIX pos <= 0) { \
510 return false; \
511 } \
512 if (SEED_TAB[(unsigned char)(MEMBER_PREFIX seq[MEMBER_PREFIX pos - 1])] == \
513 SEED_N) { \
514 MEMBER_PREFIX pos -= MEMBER_PREFIX k; \
515 return init(); \
516 } \
517 (NTHASH_CALL); \
518 --MEMBER_PREFIX pos; \
519 return true; \
520 }
