287MIBloomFilterInitializer::parse_header(const std::string& expected_signature)
288{
289 check_file_accessibility(path);
290 btllib::check_error(ifs_id_arr.fail(),
291 "MIBloomFilterInitializer: failed to open " + path);
292
293 std::string file_signature;
294 if (!check_file_signature(ifs_id_arr, expected_signature, file_signature)) {
295 log_error(std::string("File signature does not match (possibly version "
296 "mismatch) for file:\n") +
297 path + '\n' + "Expected signature:\t" + expected_signature +
298 '\n' + "File signature: \t" + file_signature);
299 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
300 }
301
302 /* Read bloom filter line by line until it sees "[HeaderEnd]"
303 * which is used to mark the end of the header section and
304 * assigns the header to a char array*/
305 std::string toml_buffer(file_signature + '\n');
306 std::string line;
307 bool header_end_found = false;
308 while (bool(std::getline(ifs_id_arr, line))) {
309 toml_buffer.append(line + '\n');
310 if (line == "[HeaderEnd]") {
311 header_end_found = true;
312 break;
313 }
314 }
315 if (!header_end_found) {
316 log_error("Pre-built multi-index Bloom filter does not have the correct "
317 "header end.");
318 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
319 }
320 for (unsigned i = 0; i < PLACEHOLDER_NEWLINES_MIBF; i++) {
321 std::getline(ifs_id_arr, line);
322 }
323
324 // Send the char array to a stringstream for the cpptoml parser to parse
325 std::istringstream toml_stream(toml_buffer);
326 cpptoml::parser toml_parser(toml_stream);
327 const auto header_config = toml_parser.parse();
328
329 // Obtain header values from toml parser and assign them to class members
330 const auto header_string =
331 file_signature.substr(1, file_signature.size() - 2); // Remove [ ]
332 return header_config->get_table(header_string);
333}

401inline void
402MIBloomFilter<T>::insert_bv(const uint64_t* hashes)
403{
404 assert(!bv_insertion_completed);
405 // check array size = hash_num
406 for (unsigned i = 0; i < hash_num; ++i) {
407 uint64_t pos = hashes[i] % bit_vector.size();
408 uint64_t* data_index = bit_vector.data() + (pos >> 6); // NOLINT
409 uint64_t bit_mask_value = (uint64_t)1 << (pos & 0x3F); // NOLINT
410 (void)(__sync_fetch_and_or(data_index, bit_mask_value) >>
411 (pos & 0x3F) & // NOLINT
412 1); // NOLINT
413 }

447inline void
448MIBloomFilter<T>::insert_id(const uint64_t* hashes, const T& id)
449{
450 assert(bv_insertion_completed && !id_insertion_completed);
451
452 uint rand = std::rand(); // NOLINT
453 for (unsigned i = 0; i < hash_num; ++i) {
454 uint64_t rank = get_rank_pos(hashes[i]);
455 uint16_t count = ++counts_array[rank];
456 T random_num = (rand ^ hashes[i]) % count;
457 if (random_num == count - 1) {
458 set_data(rank, id);
459 }
460 }

468template<typename T>
469inline void
470MIBloomFilter<T>::insert_saturation(const uint64_t* hashes, const T& id)
471{
472 assert(id_insertion_completed);
473 std::vector<uint64_t> rank_pos = get_rank_pos(hashes);
474 std::vector<T> results = get_data(rank_pos);
475 std::vector<T> replacement_ids(hash_num);
476 bool value_found = false;
477 std::vector<T> seen_set(hash_num);
478
479 for (unsigned i = 0; i < hash_num; i++) {
480 T current_result = results[i] & (btllib::MIBloomFilter<T>::ANTI_MASK &
481 btllib::MIBloomFilter<T>::ANTI_STRAND);
482 // break if ID exists
483 if (current_result == id) {
484 value_found = true;
485 break;
486 }
487 // if haven't seen before add to seen set
488 if (find(seen_set.begin(), seen_set.end(), current_result) ==
489 seen_set.end()) {
490 seen_set.push_back(current_result);
491 }
492 // if have seen before add to replacement IDs
493 else {
494 replacement_ids.push_back(current_result);
495 }
496 }
497 // if value not found try to survive
498 if (!value_found) {
499 uint64_t replacement_pos = id_array_size;
500 T min_count = std::numeric_limits<T>::min();
501 for (unsigned i = 0; i < hash_num; i++) {
502 T current_result = results[i] & btllib::MIBloomFilter<T>::ANTI_MASK;
503 if (find(replacement_ids.begin(),
504 replacement_ids.end(),
505 current_result) != replacement_ids.end()) {
506 if (min_count < counts_array[rank_pos[i]]) {
507 min_count = counts_array[rank_pos[i]];
508 replacement_pos = rank_pos[i];
509 }
510 }
511 }
512 if (replacement_pos != id_array_size) {
513 set_data(replacement_pos, id);
514 ++counts_array[replacement_pos];
515 } else {
516 set_saturated(hashes);
517 }
518 }
519}

636template<typename T>
637inline std::vector<size_t>
638MIBloomFilter<T>::get_id_occurence_count(const bool& include_saturated)
639{
640 // Ensure the bloom filter has been initialized
641 assert(bv_insertion_completed);
642
643 // Initialize a temporary vector to store counts
644 std::vector<std::atomic<size_t>> count_vec(MASK - 1);
645
646 // Iterate over the id_array in parallel, incrementing the counts in count_vec
647#pragma omp parallel for default(none) \
648 shared(id_array_size, include_saturated, id_array, count_vec)
649 for (size_t k = 0; k < id_array_size; k++) {
650 if (!include_saturated && id_array[k] > ANTI_MASK) {
651 continue;
652 }
653 count_vec[id_array[k] & ANTI_MASK].fetch_add(1);
654 }
655
656 // Convert the atomic count_vec to a non-atomic result vector,
657 // excluding trailing zeros
658 std::vector<size_t> result;
659 bool has_trailing_zeros = true;
660 for (size_t i = MASK - 2; i != SIZE_MAX; i--) {
661 if (count_vec[i] != 0) {
662 has_trailing_zeros = false;
663 }
664 if (!has_trailing_zeros) {
665 result.insert(result.begin(), count_vec[i].load());
666 }
667 }
668
669 return result;
670}
