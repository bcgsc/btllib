36inline bool
37SeqReaderGfa2Module::buffer_valid(const char* buffer, const size_t size)
38{
39 const unsigned char specs[] = { 'H', 'S', 'F', 'E', 'G', 'O', 'U' };
40
41 enum State
42 {
43 IN_ID,
44 IN_ID_TAB,
45 IN_REST,
46 IN_IGNORED
47 };
48
49 auto is_a_spec = [&](unsigned char c) {
50 bool found = false;
51 for (unsigned char spec : specs) {
52 if (c == spec) {
53 found = true;
54 break;
55 }
56 }
57 return found;
58 };
59
60 State state = is_a_spec(buffer[0]) ? IN_ID : IN_IGNORED;
61 bool has_id = false;
62 size_t current = 0;
63 unsigned char c;
64 while (current < size) {
65 c = buffer[current];
66 switch (state) {
67 case IN_ID:
68 if (!is_a_spec(c)) {
69 return false;
70 }
71 has_id = true;
72 state = IN_ID_TAB;
73 break;
74 case IN_ID_TAB:
75 if (c != '\t') {
76 return false;
77 }
78 state = IN_REST;
79 break;
80 case IN_REST:
81 if (c == '\n') {
82 if (current + 1 < size) {
83 state = is_a_spec(buffer[current + 1]) ? IN_ID : IN_IGNORED;
84 }
85 }
86 break;
87 case IN_IGNORED:
88 if (c == '\n') {
89 if (current + 1 < size) {
90 state = is_a_spec(buffer[current + 1]) ? IN_ID : IN_IGNORED;
91 }
92 }
93 break;
94 default:
95 break;
96 }
97 current++;
98 }
99
100 return has_id;
101}
