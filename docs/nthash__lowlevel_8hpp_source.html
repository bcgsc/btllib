50inline uint64_t
51srol(const uint64_t x)
52{
53 uint64_t m = ((x & 0x8000000000000000ULL) >> 30) | // NOLINT
54 ((x & 0x100000000ULL) >> 32); // NOLINT
55 return ((x << 1) & 0xFFFFFFFDFFFFFFFFULL) | m; // NOLINT
56}

67inline uint64_t
68srol(const uint64_t x, const unsigned d)
69{
70 uint64_t v = (x << d) | (x >> (64 - d)); // NOLINT
71 uint64_t y = (v ^ (v >> 33)) & // NOLINT
72 (std::numeric_limits<uint64_t>::max() >> (64 - d)); // NOLINT
73 return v ^ (y | (y << 33)); // NOLINT
74}

84inline uint64_t
85sror(const uint64_t x)
86{
87 uint64_t m = ((x & 0x200000000ULL) << 30) | ((x & 1ULL) << 32); // NOLINT
88 return ((x >> 1) & 0xFFFFFFFEFFFFFFFFULL) | m; // NOLINT
89}

251inline void
252nte64(const uint64_t bh_val,
253 const unsigned k,
254 const unsigned h,
255 uint64_t* h_val)
256{
257 uint64_t t_val;
258 h_val[0] = bh_val;
259 for (unsigned i = 1; i < h; i++) {
260 t_val = bh_val * (i ^ k * MULTISEED);
261 t_val ^= t_val >> MULTISHIFT;
262 h_val[i] = t_val;
263 }
264}

549#define NTMSM64(ROL_HANDLING, IN_HANDLING, OUT_HANDLING, ROR_HANDLING) \
550 unsigned char char_out, char_in; \
551 uint64_t fh_seed, rh_seed; \
552 unsigned i_out, i_in, i_base; \
553 for (unsigned i_seed = 0; i_seed < m; i_seed++) { \
554 ROL_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
555 for (const auto& block : seeds_blocks[i_seed]) \
556 { \
557 IN_HANDLING \
558 OUT_HANDLING \
559 fh_seed ^= MS_TAB(char_out, k - i_out); \
560 fh_seed ^= MS_TAB(char_in, k - i_in); \
561 rh_seed ^= MS_TAB(char_out & CP_OFF, i_out); \
562 rh_seed ^= MS_TAB(char_in & CP_OFF, i_in); \
563 } \
564 ROR_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
565 fh_nomonos[i_seed] = fh_seed; \
566 rh_nomonos[i_seed] = rh_seed; \
567 for (const auto& pos : seeds_monomers[i_seed]) { \
568 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1], k - 1 - pos); \
569 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1] & CP_OFF, pos); \
570 } \
571 fh_val[i_seed] = fh_seed; \
572 rh_val[i_seed] = rh_seed; \
573 i_base = i_seed * m2; \
574 h_val[i_base] = canonical(fh_seed, rh_seed); \
575 for (unsigned i_hash = 1; i_hash < m2; i_hash++) { \
576 h_val[i_base + i_hash] = h_val[i_base] * (i_hash ^ k * MULTISEED); \
577 h_val[i_base + i_hash] ^= h_val[i_base + i_hash] >> MULTISHIFT; \
578 } \
579 }
