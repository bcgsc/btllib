53inline uint64_t
54srol(const uint64_t x)
55{
56 uint64_t m = ((x & 0x8000000000000000ULL) >> 30) | // NOLINT
57 ((x & 0x100000000ULL) >> 32); // NOLINT
58 return ((x << 1) & 0xFFFFFFFDFFFFFFFFULL) | m; // NOLINT
59}
60
70inline uint64_t
71srol(const uint64_t x, const unsigned d)
72{
73 uint64_t v = (x << d) | (x >> (64 - d)); // NOLINT
74 uint64_t y = (v ^ (v >> 33)) & // NOLINT
75 (std::numeric_limits<uint64_t>::max() >> (64 - d)); // NOLINT
76 return v ^ (y | (y << 33)); // NOLINT
77}

87inline uint64_t
88sror(const uint64_t x)
89{
90 uint64_t m = ((x & 0x200000000ULL) << 30) | ((x & 1ULL) << 32); // NOLINT
91 return ((x >> 1) & 0xFFFFFFFEFFFFFFFFULL) | m; // NOLINT
92}

102inline uint64_t
103ntf64(const char* kmer_seq, unsigned k)
104{
105 uint64_t h_val = 0;
106 for (unsigned i = 0; i < k / 4; i++) {
107 h_val = srol(h_val, 4);
108 uint8_t curr_offset = 4 * i;
109 uint8_t tetramer_loc =
110 64 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset]] + // NOLINT
111 16 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 1]] + // NOLINT
112 4 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 2]] +
113 CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 3]];
114 h_val ^= TETRAMER_TAB[tetramer_loc];
115 }
116 unsigned remainder = k % 4;
117 h_val = srol(h_val, remainder);
118 if (remainder == 3) {
119 uint8_t trimer_loc =
120 16 * CONVERT_TAB[(unsigned char)kmer_seq[k - 3]] + // NOLINT
121 4 * CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
122 CONVERT_TAB[(unsigned char)kmer_seq[k - 1]];
123 h_val ^= TRIMER_TAB[trimer_loc];
124 } else if (remainder == 2) {
125 uint8_t dimer_loc = 4 * CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
126 CONVERT_TAB[(unsigned char)kmer_seq[k - 1]];
127 h_val ^= DIMER_TAB[dimer_loc];
128 } else if (remainder == 1) {
129 h_val ^= SEED_TAB[(unsigned char)kmer_seq[k - 1]];
130 }
131 return h_val;
132}

143inline uint64_t
144ntr64(const char* kmer_seq, unsigned k)
145{
146 uint64_t h_val = 0;
147 unsigned remainder = k % 4;
148 if (remainder == 3) {
149 uint8_t trimer_loc =
150 16 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 1]] + // NOLINT
151 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
152 RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 3]];
153 h_val ^= TRIMER_TAB[trimer_loc];
154 } else if (remainder == 2) {
155 uint8_t dimer_loc = 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 1]] +
156 RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 2]];
157 h_val ^= DIMER_TAB[dimer_loc];
158 } else if (remainder == 1) {
159 h_val ^= SEED_TAB[(unsigned char)kmer_seq[k - 1] & CP_OFF];
160 }
161 for (unsigned i = 0; i < k / 4; i++) {
162 h_val = srol(h_val, 4);
163 uint8_t curr_offset = 4 * (k / 4 - i) - 1;
164 uint8_t tetramer_loc =
165 64 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset]] + // NOLINT
166 16 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 1]] + // NOLINT
167 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 2]] +
168 RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 3]];
169 h_val ^= TETRAMER_TAB[tetramer_loc];
170 }
171 return h_val;
172}

356inline void
357nte64(uint64_t bh_val, unsigned k, unsigned h, uint64_t* h_val)
358{
359 uint64_t t_val;
360 h_val[0] = bh_val;
361 for (unsigned i = 1; i < h; i++) {
362 t_val = bh_val * (i ^ k * MULTISEED);
363 t_val ^= t_val >> MULTISHIFT;
364 h_val[i] = t_val;
365 }
366}

467ntc64(const char* kmer_seq, unsigned k, uint64_t& h_val, unsigned& loc_n)
468{
469 h_val = 0;
470 loc_n = 0;
471 uint64_t fh_val = 0, rh_val = 0;

481 rh_val ^= SEED_TAB[(unsigned char)kmer_seq[i] & CP_OFF];
482 }
483 h_val = canonical(fh_val, rh_val);

695inline uint64_t
696mask_hash(uint64_t& fk_val,
697 uint64_t& rk_val,
698 const char* seed_seq,
699 const char* kmer_seq,
700 unsigned k)
701{
702 uint64_t fs_val = fk_val, rs_val = rk_val;
703 for (unsigned i = 0; i < k; i++) {
704 if (seed_seq[i] != '1') {
705 fs_val ^= MS_TAB((unsigned char)kmer_seq[i], k - 1 - i);
706 rs_val ^= MS_TAB((unsigned char)kmer_seq[i] & CP_OFF, i);
707 }
708 }
709 return canonical(fs_val, rs_val);
710}

726inline void
727sub_hash(uint64_t fh_val,
728 uint64_t rh_val,
729 const char* kmer_seq,
730 const std::vector<unsigned>& positions,
731 const std::vector<unsigned char>& new_bases,
732 unsigned k,
733 unsigned m,
734 uint64_t* h_val)
735{
736 uint64_t b_val = 0;
737
738 for (size_t i = 0; i < positions.size(); i++) {
739 const auto pos = positions[i];
740 const auto new_base = new_bases[i];
741
742 fh_val ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
743 fh_val ^= MS_TAB(new_base, k - 1 - pos);
744
745 rh_val ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
746 rh_val ^= MS_TAB(new_base & CP_OFF, pos);
747 }
748
749 b_val = canonical(fh_val, rh_val);
750 nte64(b_val, k, m, h_val);
751}

777inline bool
778ntmsm64(const char* kmer_seq,
779 const std::vector<SpacedSeedBlocks>& seeds_blocks,
780 const std::vector<SpacedSeedMonomers>& seeds_monomers,
781 unsigned k,
782 unsigned m,
783 unsigned m2,
784 uint64_t* fh_nomonos,
785 uint64_t* rh_nomonos,
786 uint64_t* fh_val,
787 uint64_t* rh_val,
788 unsigned& loc_n,
789 uint64_t* h_val)
790{
791 unsigned i_base;
792 uint64_t fh_seed, rh_seed;
793 for (unsigned i_seed = 0; i_seed < m; i_seed++) {
794 fh_seed = 0;
795 rh_seed = 0;
796
797 for (const auto& block : seeds_blocks[i_seed]) {
798 for (unsigned pos = block[0]; pos < block[1]; pos++) {
799 if (kmer_seq[pos] == SEED_N) {
800 loc_n = pos;
801 return false;
802 }
803 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
804 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
805 }
806 }
807 fh_nomonos[i_seed] = fh_seed;
808 rh_nomonos[i_seed] = rh_seed;
809 for (unsigned pos : seeds_monomers[i_seed]) {
810 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
811 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
812 }
813 fh_val[i_seed] = fh_seed;
814 rh_val[i_seed] = rh_seed;
815 i_base = i_seed * m2;
816 h_val[i_base] = canonical(fh_seed, rh_seed);
817 for (unsigned i_hash = 1; i_hash < m2; i_hash++) {
818 h_val[i_base + i_hash] = h_val[i_base] * (i_hash ^ k * MULTISEED);
819 h_val[i_base + i_hash] ^= h_val[i_base + i_hash] >> MULTISHIFT;
820 }
821 }
822 return true;
823}

826 unsigned char char_out, char_in; \
827 uint64_t fh_seed, rh_seed; \
828 unsigned i_out, i_in, i_base; \
829 for (unsigned i_seed = 0; i_seed < m; i_seed++) { \
830 ROL_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
831 for (const auto& block : seeds_blocks[i_seed]) \
832 { \
833 IN_HANDLING \
834 OUT_HANDLING \
835 fh_seed ^= MS_TAB(char_out, k - i_out); \
836 fh_seed ^= MS_TAB(char_in, k - i_in); \
837 rh_seed ^= MS_TAB(char_out & CP_OFF, i_out); \
838 rh_seed ^= MS_TAB(char_in & CP_OFF, i_in); \
839 } \
840 ROR_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
841 fh_nomonos[i_seed] = fh_seed; \
842 rh_nomonos[i_seed] = rh_seed; \
843 for (const auto& pos : seeds_monomers[i_seed]) { \
844 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1], k - 1 - pos); \
845 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1] & CP_OFF, pos); \

847 fh_val[i_seed] = fh_seed; \
848 rh_val[i_seed] = rh_seed; \
849 i_base = i_seed * m2; \
850 h_val[i_base] = canonical(fh_seed, rh_seed); \
851 for (unsigned i_hash = 1; i_hash < m2; i_hash++) { \
852 h_val[i_base + i_hash] = h_val[i_base] * (i_hash ^ k * MULTISEED); \
853 h_val[i_base + i_hash] ^= h_val[i_base + i_hash] >> MULTISHIFT; \
854 } \
