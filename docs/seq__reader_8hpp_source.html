318 {
319  buffer.start = 0;
320  char last = buffer.end > 0 ? buffer.data[buffer.end - 1] : char(0);
321  buffer.end = 0;
322  do {
323  buffer.end += fread(buffer.data.data() + buffer.end,

327  } while (buffer.end < buffer.data.size() && !bool(std::feof(source)));
328 
329  if (bool(std::feof(source)) && !buffer.eof_newline_inserted) {
330  if (buffer.end < buffer.data.size()) {
331  if ((buffer.end == 0 && last != '\n') ||
332  (buffer.end > 0 && buffer.data[buffer.end - 1] != '\n')) {
333  buffer.data[buffer.end++] = '\n';

335  buffer.eof_newline_inserted = true;

532 inline void
533 SeqReader::start_processors()
534 {
535  processor_threads.reserve(threads);
536  for (unsigned i = 0; i < threads; i++) {
537  processor_threads.push_back(
538  std::unique_ptr<std::thread>(new std::thread([this]() {
539  decltype(cstring_queue)::Block records_in(block_size);
540  decltype(output_queue)::Block records_out(block_size);
541  for (;;) {
542  cstring_queue.read(records_in);
543  for (size_t i = 0; i < records_in.count; i++) {
544  records_out.data[i].seq = std::string(
545  records_in.data[i].seq, records_in.data[i].seq.size());
546  auto& seq = records_out.data[i].seq;
547  while (!seq.empty() && (seq.back() == '\r' || seq.back() == '\n')) {
548  seq.pop_back();
549  }
550 
551  records_out.data[i].qual = std::string(
552  records_in.data[i].qual, records_in.data[i].qual.size());
553  auto& qual = records_out.data[i].qual;
554  while (!qual.empty() &&
555  (qual.back() == '\r' || qual.back() == '\n')) {
556  qual.pop_back();
557  }
558 
559  char *first_whitespace = nullptr, *last_whitespace = nullptr;
560  for (size_t j = 0; j < records_in.data[i].header.size(); j++) {
561  if (bool(std::isspace(records_in.data[i].header[j]))) {
562  if (first_whitespace == nullptr) {
563  first_whitespace = records_in.data[i].header + j;
564  }
565  last_whitespace = records_in.data[i].header + j;
566  } else if (last_whitespace != nullptr) {
567  break;
568  }
569  }
570  size_t id_start =
571  (format == Format::FASTA || format == Format::FASTQ) ? 1 : 0;
572 
573  if (first_whitespace == nullptr) {
574  records_out.data[i].id =
575  std::string(records_in.data[i].header + id_start,
576  records_in.data[i].header.size() - id_start);
577  records_out.data[i].comment = "";
578  } else {
579  records_out.data[i].id = std::string(
580  records_in.data[i].header + id_start,
581  first_whitespace - records_in.data[i].header - id_start);
582  records_out.data[i].comment = std::string(
583  last_whitespace + 1,
584  records_in.data[i].header.size() -
585  (last_whitespace - records_in.data[i].header) - 1);
586  }
587  records_in.data[i].header.clear();
588 
589  auto& id = records_out.data[i].id;
590  auto& comment = records_out.data[i].comment;
591  while (!id.empty() && (id.back() == '\r' || id.back() == '\n')) {
592  id.pop_back();
593  }
594  while (!comment.empty() &&
595  (comment.back() == '\r' || comment.back() == '\n')) {
596  comment.pop_back();
597  }
598 
599  if (trim_masked()) {
600  const auto len = seq.length();
601  size_t trim_start = 0, trim_end = seq.length();
602  while (trim_start <= len && bool(islower(seq[trim_start]))) {
603  trim_start++;
604  }
605  while (trim_end > 0 && bool(islower(seq[trim_end - 1]))) {
606  trim_end--;
607  }
608  seq.erase(trim_end);
609  seq.erase(0, trim_start);
610  if (!qual.empty()) {
611  qual.erase(trim_end);
612  qual.erase(0, trim_start);
613  }
614  }
615  if (fold_case()) {
616  for (auto& c : seq) {
617  char old = c;
618  c = CAPITALS[(unsigned char)(c)];
619  if (!bool(c)) {
620  log_error(std::string("A sequence contains invalid "
621  "IUPAC character: ") +
622  old);
623  std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
624  }
625  }
626  }
627  records_out.data[i].num = records_in.num * block_size + i;
628  }
629  records_out.count = records_in.count;
630  records_out.current = records_in.current;
631  records_out.num = records_in.num;
632  if (records_out.count == 0) {
633  if (records_out.num == dummy_block_num) {
634  output_queue.write(records_out);
635  }
636  break;
637  }
638  output_queue.write(records_out);
639  }
640  })));
641  }

645 SeqReader::read()
646 {
647  if (ready_records_owners()[id % MAX_SIMULTANEOUS_SEQREADERS] != id) {
648  ready_records_array()[id % MAX_SIMULTANEOUS_SEQREADERS] =
649  std::unique_ptr<decltype(output_queue)::Block>(
650  new decltype(output_queue)::Block(block_size));
651  ready_records_owners()[id % MAX_SIMULTANEOUS_SEQREADERS] = id;
652  }

654  *(ready_records_array()[id % MAX_SIMULTANEOUS_SEQREADERS]);
655  if (ready_records.count <= ready_records.current) {
656  output_queue.read(ready_records);
657  if (ready_records.count <= ready_records.current) {
658  close();
659  ready_records = decltype(output_queue)::Block(block_size);
660  return Record();
661  }
662  }
663  return std::move(ready_records.data[ready_records.current++]);
