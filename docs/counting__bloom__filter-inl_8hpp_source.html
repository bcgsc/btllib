57template<typename T>
58inline void
59CountingBloomFilter<T>::insert(const uint64_t* hashes, T min_val)
60{
61 // Update flag to track if increment is done on at least one counter
62 bool update_done = false;
63 T new_val, tmp_min_val;
64 while (true) {
65 new_val = min_val + 1;
66 for (size_t i = 0; i < hash_num; ++i) {
67 tmp_min_val = min_val;
68 update_done |= array[hashes[i] % array_size].compare_exchange_strong(
69 tmp_min_val, new_val);
70 }
71 if (update_done) {
72 break;
73 }
74 min_val = contains(hashes);
75 if (min_val == std::numeric_limits<T>::max()) {
76 break;
77 }
78 }
79}

200template<typename T>
201inline uint64_t
202CountingBloomFilter<T>::get_pop_cnt(const T threshold) const
203{
204 uint64_t pop_cnt = 0;
205// OpenMP make up your mind man. Using default(none) here causes errors on
206// some compilers and not others.
207// NOLINTNEXTLINE(openmp-use-default-none,-warnings-as-errors)
208#pragma omp parallel for reduction(+ : pop_cnt)
209 for (size_t i = 0; i < array_size; ++i) {
210 if (array[i] >= threshold) {
211 ++pop_cnt;
212 }
213 }
214 return pop_cnt;
215}

266inline void
267CountingBloomFilter<T>::save(const std::string& path)
268{
269 /* Initialize cpptoml root table
270 Note: Tables and fields are unordered
271 Ordering of table is maintained by directing the table
272 to the output stream immediately after completion */
273 auto root = cpptoml::make_table();
274
275 /* Initialize bloom filter section and insert fields
276 and output to ostream */
277 auto header = cpptoml::make_table();
278 header->insert("bytes", get_bytes());
279 header->insert("hash_num", get_hash_num());
280 if (!hash_fn.empty()) {
281 header->insert("hash_fn", hash_fn);
282 }
283 header->insert("counter_bits", size_t(sizeof(array[0]) * CHAR_BIT));
284 std::string header_string = COUNTING_BLOOM_FILTER_SIGNATURE;
285 header_string =
286 header_string.substr(1, header_string.size() - 2); // Remove [ ]
287 root->insert(header_string, header);
288
289 BloomFilter::save(
290 path, *root, (char*)array.get(), array_size * sizeof(array[0]));
291}
