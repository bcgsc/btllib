778 * Assumes min_count is not std::numeric_limits<T>::max()
779 */
780template<typename T>
781inline void
782CountingBloomFilter<T>::insert(const uint64_t* hashes, T min_val)
783{
784 // Update flag to track if increment is done on at least one counter
785 bool update_done = false;
786 T new_val, tmp_min_val;
787 while (true) {
788 new_val = min_val + 1;
789 for (size_t i = 0; i < hash_num; ++i) {
790 tmp_min_val = min_val;
791 update_done |= array[hashes[i] % array_size].compare_exchange_strong(
792 tmp_min_val, new_val);
793 }
794 if (update_done) {
795 break;
796 }
797 min_val = contains(hashes);
798 if (min_val == std::numeric_limits<T>::max()) {
799 break;
800 }
801 }
802}

923template<typename T>
924inline uint64_t
925CountingBloomFilter<T>::get_pop_cnt(const T threshold) const
926{
927 uint64_t pop_cnt = 0;
928 // OpenMP make up your mind man. Using default(none) here causes errors on
929 // some compilers and not others.
930 // NOLINTNEXTLINE(openmp-use-default-none,-warnings-as-errors)
931#pragma omp parallel for reduction(+ : pop_cnt)
932 for (size_t i = 0; i < array_size; ++i) {
933 if (array[i] >= threshold) {
934 ++pop_cnt;
935 }
936 }
937 return pop_cnt;
938}

988template<typename T>
989inline void
990CountingBloomFilter<T>::save(const std::string& path)
991{
992 /* Initialize cpptoml root table
993 Note: Tables and fields are unordered
994 Ordering of table is maintained by directing the table
995 to the output stream immediately after completion */
996 auto root = cpptoml::make_table();
997
998 /* Initialize bloom filter section and insert fields
999 and output to ostream */
1000 auto header = cpptoml::make_table();
1001 header->insert("bytes", get_bytes());
1002 header->insert("hash_num", get_hash_num());
1003 if (!hash_fn.empty()) {
1004 header->insert("hash_fn", hash_fn);
1005 }
1006 header->insert("counter_bits", size_t(sizeof(array[0]) * CHAR_BIT));
1007 std::string header_string = COUNTING_BLOOM_FILTER_SIGNATURE;
1008 header_string =
1009 header_string.substr(1, header_string.size() - 2); // Remove [ ]
1010 root->insert(header_string, header);
1011
1012 BloomFilter::save(
1013 path, *root, (char*)array.get(), array_size * sizeof(array[0]));
1014}
